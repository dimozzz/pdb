\task{ Бесконечная цепочка, изображённая на рисунке, содержит
  резисторы сопротивлением $R$ и $r$. Чему равно сопротивление
  измеренное между двумя узлами, между которыми находится $n$
  резисторов с сопротивлением $r$? }

\begin{center}
  \begin{tikzpicture}[circuit ee IEC]
    % рисуем сетку контактов
    \foreach \contact in {0,1,...,3}
    {
      \node[contact] (top contact \contact) at (2*\contact,2) {};
      \node[contact] (bottom contact \contact) at (2*\contact,0) {};
    }
    % рисуем резисторы r
    \foreach \i in {0,1,2} { \draw[very thick] (2*\i,2)
      to[resistor={info=$r$}] ++(2,0); }
    
    % рисуем резисторы R
    \foreach \i in {0,1,2,3} { \draw[very thick,circuit symbol
      unit=0.35cm] (2*\i,2) to[resistor={info=$R$}] ++(0,-2); }
    \draw[very thick] (-0.5,0) -- (6.5,0) (-0.5,2) -- (0,2) (6,2) -- (6.5,2);
  \end{tikzpicture}
\end{center}
